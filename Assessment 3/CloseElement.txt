package com;

class tree {
    int data;
    tree left, right;
    tree(int data) {
        this.data = data;
    }
}

public class CloseElement {

    
    public static tree findLCA(tree root, int n1, int n2) {
        if (root == null) 
        	return null;

      
        if (root.data == n1 || root.data == n2) 
        	return root;

        tree leftLCA = findLCA(root.left, n1, n2);
        tree rightLCA = findLCA(root.right, n1, n2);

        if (leftLCA != null && rightLCA != null)
        	return root;

        return (leftLCA != null) ? leftLCA : rightLCA;
    }

    public static void main(String[] args) {
        
        tree root = new tree(1);
        root.left = new tree(2);
        root.right = new tree(3);
        root.left.left = new tree(4);
        root.left.right = new tree(5);
        tree lca1 = findLCA(root, 4, 5);
        System.out.println("LCA of 4 and 5 is: " + lca1.data);
    }
}
