  <script>
    // Very simple JS for wireframe behavior

    // Login handling (fake)
    const loginPage = document.getElementById('login-page');
    const dashboard = document.getElementById('dashboard');
    const loginBtn = document.getElementById('btn-login');
    const loginError = document.getElementById('login-error');

    loginBtn.addEventListener('click', function() {
      const username = document.getElementById('admin-username').value.trim();
      const password = document.getElementById('admin-password').value.trim();

      // Simple front-end check: require both fields
      if (!username || !password) {
        loginError.style.display = 'block';
      } else {
        loginError.style.display = 'none';
        loginPage.classList.add('hidden');
        dashboard.classList.remove('hidden');
      }
    });

    // Simple tab switching
    const tabButtons = document.querySelectorAll('nav button');
    const tabSections = document.querySelectorAll('.tab-section');

    tabButtons.forEach(btn => {
      btn.addEventListener('click', () => {
        const target = btn.getAttribute('data-tab');

        tabButtons.forEach(b => b.classList.remove('active-tab'));
        btn.classList.add('active-tab');

        tabSections.forEach(sec => {
          if (sec.id === target) {
            sec.classList.remove('hidden');
          } else {
            sec.classList.add('hidden');
          }
        });
      });
    });

    // Fake user data for View All Users wireframe
    const users = [
      { id: 1, name: 'Ravi Kumar', email: 'ravi.kumar@example.com', cardType: 'Gold', status: 'pending' },
      { id: 2, name: 'Priya Sharma', email: 'priya.sharma@example.com', cardType: 'Platinum', status: 'pending' },
      { id: 3, name: 'Anita Singh', email: 'anita.singh@example.com', cardType: 'Silver', status: 'approved' },
      { id: 4, name: 'Rahul Verma', email: 'rahul.verma@example.com', cardType: 'Gold', status: 'approved' },
      { id: 5, name: 'Test User', email: 'test.user@example.com', cardType: 'Silver', status: 'rejected' }
    ];

    const usersTbody = document.getElementById('users-tbody');
    const usersEmpty = document.getElementById('users-empty');
    const filterStatus = document.getElementById('filter-status');
    const searchUser = document.getElementById('search-user');

    function renderUsers() {
      const statusValue = filterStatus.value;
      const searchValue = searchUser.value.trim().toLowerCase();

      const filtered = users.filter(u => {
        let statusMatch = (statusValue === 'all') || (u.status === statusValue);
        let searchMatch =
          !searchValue ||
          u.name.toLowerCase().includes(searchValue) ||
          u.email.toLowerCase().includes(searchValue);

        return statusMatch && searchMatch;
      });

      usersTbody.innerHTML = '';

      if (filtered.length === 0) {
        usersEmpty.style.display = 'block';
        return;
      } else {
        usersEmpty.style.display = 'none';
      }

      filtered.forEach((u, index) => {
        const tr = document.createElement('tr');

        const statusClassMap = {
          pending: 'status-pending',
          approved: 'status-approved',
          rejected: 'status-rejected'
        };
        const statusLabel = u.status.charAt(0).toUpperCase() + u.status.slice(1);

        tr.innerHTML = `
          <td>${index + 1}</td>
          <td>${u.name}</td>
          <td>${u.email}</td>
          <td>${u.cardType}</td>
          <td><span class="status-badge ${statusClassMap[u.status]}">${statusLabel}</span></td>
        `;

        usersTbody.appendChild(tr);
      });
    }

    filterStatus.addEventListener('change', renderUsers);
    searchUser.addEventListener('input', renderUsers);

    // Initial render
    renderUsers();
  </script>
